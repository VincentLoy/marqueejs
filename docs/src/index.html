Modern, lightweight and flexible marquee solution for your web applications

Fully Customizable

Multiple directions, dynamic content updates, custom speeds, and flexible styling options.

Modern API

Usage

Import and initialize Marquee JS 

import { marqueejs } from 'marqueejs';

// or just marqueejs('selector', options); if you'll never use the methods
let marqueeInstance = marqueejs('.selector', {
    // Core animation options
    speed: 100,
    direction: 'left',
    startAfter: 1000,
    pauseOnHover: true,

    // Layout options
    gap: 20,
    containerHeight: 400,
    heightSecurityMargin: 10,

    // Content management
    contentList: [
        '<div class="item">Item 1</div>',
        '<div class="item">Item 2</div>',
        '<div class="item">Item 3</div>',
    ],
    keepOriginalContent: true,
    randomize: false,

    // Clone management
    cloneCount: 'auto', // or number between 1-15

    // Separator configuration
    separator: '•',
    separatorStyles: {
        fontSize: '1.2em',
        color: '#666',
        marginTop: '-2px'
    },

    // Content validation
    contentValidation: {
        maxLength: 1000,
        forbiddenTags: ['script', 'iframe'],
        forbiddenAttributes: ['onerror', 'onload']
    }
});
